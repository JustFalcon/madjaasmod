country_event = {
	id = madyas_events.2
	title = "madyas_events.2.t"
	desc = "madyas_events.2.d"
	picture = EMPIRE_eventPicture
	

	trigger = { }

	immediate = {
		hidden_effect = {
			ROOT = {
				save_event_target_as = mas_whois_opinion
			}
			every_country = {
				limit = {
					is_emperor_of_china = yes
				}
				save_event_target_as = chinese_emperor
			}
		}
	}


	is_triggered_only = yes
	
	option = {
		name = "madyas_events.2.a" ## Madja-as Forward Base (Keep)
		event_target:chinese_emperor = {
			country_event = {
				id = madyas_events.3
				random = 3
				days = 3
			}
		}
		trigger = {
			NOT = {
				is_subject_of_type_with_overlord = {
					who = event_target:chinese_emperor
					type = tributary_state
				}
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = "madyas_events.2.b" ## Madja-as Hand Over Formosa to the Emperor (Hand over for free)
		event_target:chinese_emperor = {
			country_event = {
				id = madyas_events.4
				random = 3
				days = 3
			}
		}
		ai_chance = {
			factor = 0
		}
	}
	option = {
		name = "madyas_events.2.c" ## Madja-as Hand Over Formosa for a Price (Over Hand over for ducats)
		event_target:chinese_emperor = {
			country_event = {
				id = madyas_events.5
				random = 3
				days = 3
			}
		}
		trigger = {
			NOT = {
				is_subject_of_type_with_overlord = {
					who = event_target:chinese_emperor
					type = tributary_state
				}
			}
		}
		ai_chance = {
			factor = 1
		}
	}
}
